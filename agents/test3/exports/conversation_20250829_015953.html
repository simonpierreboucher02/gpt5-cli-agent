python import pygame import random import sys def main(): pygame.init() # Game settings WIDTH, HEIGHT = 600, 600 GRID = 20 # size of a single grid cell in pixels COLS = WIDTH // GRID ROWS = HEIGHT // GRID screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Snake") clock = pygame.time.Clock() # Colors WHITE = (255, 255, 255) BLACK = (0, 0, 0) GREEN = (0, 255, 0) DARK_GREEN = (0, 180, 0) RED = (255, 0, 0) font = pygame.font.SysFont(None, 36) def spawn_food(snake): while True: pos = (random.randrange(COLS), random.randrange(ROWS)) if pos not in snake: return pos # Initial state snake = [(COLS // 2, ROWS // 2)] dx, dy = 1, 0 next_dx, next_dy = dx, dy food = spawn_food(snake) score = 0 MOVE_DELAY = 140 # ms per move; decreases as you grow timer = 0 running = True game_over = False while running: delta = clock.tick(60) # limit to ~60 FPS for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.KEYDOWN: ndx, ndy = None, None if event.key in (pygame.K_UP, pygame.K_w): ndx, ndy = 0, -1 elif event.key in (pygame.K_DOWN, pygame.K_s): ndx, ndy = 0, 1 elif event.key in (pygame.K_LEFT, pygame.K_a): ndx, ndy = -1, 0 elif event.key in (pygame.K_RIGHT, pygame.K_d): ndx, ndy = 1, 0 elif event.key == pygame.K_r and game_over: # Restart the game snake = [(COLS // 2, ROWS // 2)] dx, dy = 1, 0 next_dx, next_dy = dx, dy food = spawn_food(snake) score = 0 MOVE_DELAY = 140 timer = 0 game_over = False if ndx is not None: # Prevent reversing into itself if (ndx, ndy) != (-dx, -dy): next_dx, next_dy = ndx, ndy if not game_over: timer += delta if timer >= MOVE_DELAY: timer -= MOVE_DELAY dx, dy = next_dx, next_dy head = snake[-1] new_head = (head[0] + dx, head[1] + dy) # Wall collision if not (0 <= new_head[0] < COLS and 0 <= new_head[1] < ROWS): game_over = True # Self collision elif new_head in snake: game_over = True else: snake.append(new_head) if new_head == food: score += 1 food = spawn_food(snake) # Slightly speed up as you grow MOVE_DELAY = max(60, MOVE_DELAY - 1) else: snake.pop(0) # Rendering screen.fill((30, 30, 30)) # Draw food fx, fy = food pygame.draw.rect(screen, RED, (fx * GRID, fy * GRID, GRID, GRID)) # Draw snake for i, (sx, sy) in enumerate(snake): color = GREEN if i == len(snake) - 1 else DARK_GREEN pygame.draw.rect(screen, color, (sx * GRID + 1, sy * GRID + 1, GRID - 2, GRID - 2)) # Score score_surf = font.render("Score: {}".format(score), True, WHITE) screen.blit(score_surf, (10, 10)) if game_over: over_surf = font.render("Game Over! Press R to restart", True, WHITE) rect = over_surf.get_rect(center=(WIDTH // 2, HEIGHT // 2)) screen.blit(over_surf, rect) pygame.display.flip() pygame.quit() sys.exit() if __name__ == "__main__": main()
